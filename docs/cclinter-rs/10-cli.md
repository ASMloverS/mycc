# 10 - CLI 接口

## 命令行参数

```rust
use clap::Parser;

#[derive(Parser)]
#[command(
    name = "cclinter-rs",
    version,
    about = "Static code checker for C/C++ (cpplint Rust port)"
)]
struct Cli {
    /// Verbosity level (0-5)
    #[arg(short, long, default_value = "1")]
    verbose: u8,

    /// Output format: emacs|vs7|eclipse|junit|sed|gsed
    #[arg(long, default_value = "emacs")]
    output: String,

    /// Category filters (comma-separated, e.g. -x,+y)
    #[arg(long)]
    filter: Option<String>,

    /// Error counting mode: total|toplevel|detailed
    #[arg(long, default_value = "total")]
    counting: String,

    /// Root directory for header guard derivation
    #[arg(long)]
    root: Option<String>,

    /// Repository root directory
    #[arg(long)]
    repository: Option<String>,

    /// Allowed line length
    #[arg(long, default_value_t = 80)]
    linelength: usize,

    /// Header file extensions
    #[arg(long)]
    headers: Option<String>,

    /// Valid file extensions
    #[arg(long)]
    extensions: Option<String>,

    /// Include order rule: default|standardcfirst
    #[arg(long, default_value = "default")]
    includeorder: String,

    /// Recursively search directories
    #[arg(long)]
    recursive: bool,

    /// Exclude paths (supports glob patterns, can repeat)
    #[arg(long)]
    exclude: Vec<String>,

    /// Suppress non-error output
    #[arg(long)]
    quiet: bool,

    /// Auto-fix issues (trailing whitespace, CRLF, UTF-8 BOM, block comments)
    #[arg(long)]
    fix: bool,

    /// Generate default cclinter-rs.toml in current directory
    #[arg(long)]
    init: bool,

    /// Files or directories to lint
    files: Vec<String>,
}
```

## 主流程

```rust
fn main() -> ExitCode {
    let cli = Cli::parse();

    // 1. --init: 生成默认配置文件
    if cli.init {
        return generate_default_config();
    }

    // 2. 加载配置
    let config = Config::load(&current_dir())
        .ok()
        .flatten()
        .unwrap_or_default();

    // 3. CLI 覆盖配置 (消费 config，返回新实例)
    let config = config.with_cli_overrides(&cli);

    // 4. 文件发现
    let files = discover_files(
        &cli.files,
        config.recursive(),
        &config.exclude_patterns(),
        &config.valid_extensions(),
    );

    if files.is_empty() {
        eprintln!("No files to lint.");
        return ExitCode::FAILURE;
    }

    // 5. 并行 lint (先 lint 获取原始文件的问题报告)
    let all_violations: Vec<Violation> = files
        .par_iter()
        .flat_map(|file| process_file(file, &config))
        .collect();

    // 6. 输出 (显示原始文件的问题)
    let error_count = all_violations.len();
    let formatter = OutputFormatter::new(&config.effective_output());
    for v in &all_violations {
        formatter.print(v);
    }

    // 7. 统计
    if !config.quiet || error_count > 0 {
        print_summary(&all_violations, &config);
    }

    // 8. Auto-fix (在输出报告之后执行修改)
    //    语义: 用户先看到原始问题，再执行修复
    //    若只想查看问题而不修复，不传 --fix 即可
    if cli.fix {
        let engine = FixEngine::from_config(&config);
        for file in &files {
            match engine.fix_file(file) {
                Ok(report) if report.changed => {
                    eprintln!("Fixed: {}", report.path.display());
                }
                Err(e) => {
                    eprintln!("fix error: {}: {}", file.display(), e);
                }
                _ => {}
            }
        }
    }

    if error_count > 0 {
        ExitCode::FAILURE
    } else {
        ExitCode::SUCCESS
    }
}
```

### --fix 执行顺序说明

**设计**: lint → 输出报告 → fix。用户先看到原始文件的问题报告，随后 fix 修改磁盘文件。理由：
- 用户可评估问题后再决定是否重新运行 `--fix`
- lint 报告反映原始代码状态，不被 fix 后的干净结果掩盖
- 若用户想验证 fix 结果，可重新运行 `cclinter-rs`（第二次应无错误）

## 文件发现

```rust
fn discover_files(
    paths: &[String],
    recursive: bool,
    excludes: &[String],
    extensions: &HashSet<String>,
) -> Vec<PathBuf> {
    let mut files = Vec::new();

    for path in paths {
        let p = PathBuf::from(path);
        if p.is_dir() {
            if recursive {
                // walkdir, filter by extension and exclude patterns
                for entry in walk_dir(&p) {
                    if is_valid_file(&entry, extensions) && !is_excluded(&entry, excludes) {
                        files.push(entry);
                    }
                }
            }
        } else if p.is_file() {
            if is_valid_file(&p, extensions) && !is_excluded(&p, excludes) {
                files.push(p);
            }
        }
    }

    files.sort();
    files.dedup();
    files
}
```

## --init 输出

生成默认 `cclinter-rs.toml`：

- 若 `cclinter-rs.toml` 已存在 → 打印错误提示并返回退出码 2，不覆盖
- 若不存在 → 生成默认配置文件

```toml
# cclinter-rs.toml - generated by cclinter-rs --init

verbose = 1
line_length = 80
output = "emacs"
recursive = false
quiet = false
include_order = "default"

[filters]
add = ["-build/include_alpha"]

[extensions]
headers = ["h", "hh", "hpp", "hxx", "h++", "cuh"]
sources = ["c", "cc", "cpp", "cxx", "c++", "cu"]

[fix]
trailing_whitespace = true
utf8_bom = true
crlf = true
block_comments = true
```

## 退出码

| 退出码 | 含义 |
|--------|------|
| 0 | 无错误 |
| 1 | 有 lint 错误 |
| 2 | CLI 参数错误 / 配置文件错误 / --init 文件已存在 |
